/**
 * 全局排序逻辑：控制页面上所有 class="anime-grid" 的容器
 */
function handleSortChange(selectElement) {
    const order = selectElement.value; // 'desc' or 'asc'
    const allGrids = document.getElementsByClassName('anime-grid');

    // 遍历每一个网格区域进行排序
    Array.from(allGrids).forEach(grid => {
        const cards = Array.from(grid.getElementsByClassName('anime-card'));

        cards.sort((a, b) => {
            const yearA = parseInt(a.getAttribute('data-year')) || 0;
            const yearB = parseInt(b.getAttribute('data-year')) || 0;
            
            if (order === 'desc') {
                return yearB - yearA; // 降序：2025 -> 2009
            } else {
                return yearA - yearB; // 升序：2009 -> 2025
            }
        });

        // 重新插入DOM以改变顺序
        cards.forEach(card => grid.appendChild(card));
    });
}

// 页面加载完成后，默认执行一次降序排序
document.addEventListener('DOMContentLoaded', () => {
    handleSortChange(document.getElementById('sortOrder'));
});
</script>
